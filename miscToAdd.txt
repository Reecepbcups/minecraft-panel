import platform
def checkIfSystemIsUbuntuOrArch() -> str:
    distr = platform.dist()[0]
    if distr in ['Ubuntu', "Debian"]:
        return "apt-get install"
    elif distr in ['Arch']:
        return "pacman -S"
    else:
        return 'N/A'

distr = checkIfSystemIsUbuntuOrArch()
print(distr)
